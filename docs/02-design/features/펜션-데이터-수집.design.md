model RawCrawlResult {

  @@map("raw_crawl_results")

### 3.2 공통 타입 (`types.ts`)
```typescript
export interface RawPropertyData {
  source: string
  sourceId?: string
  name: string
  address?: string
  phone?: string
  lat?: number
  lng?: number
  rating?: number        // 원본 평점 (스케일 다름)
  reviewCount?: number
  reviews?: Array<{ text: string; rating: number; date: string }>
  amenities?: string[]
  thumbnailUrl?: string
  extraData?: Record<string, unknown>
}

export interface NormalizedData extends RawPropertyData {
  ratingNormalized: number  // 5점 만점 정규화
}
```

### 3.3 VINTEE Score 계산 (`score-calculator.ts`)
```typescript
export function calculateVinteeScore(sources: NormalizedData[]): {
  score: number
  breakdown: ScoreBreakdown
} {
  // 가중치 정의
  const WEIGHTS = {
    naver:       0.30,
    google:      0.30,
    reviewVolume: 0.20,
    yanolja:     0.10,
    yeogi:       0.10,
  }

  // 소스별 평점 가중합
  let weightedSum = 0
  let totalWeight = 0

  for (const source of sources) {
    const w = WEIGHTS[source.source as keyof typeof WEIGHTS] ?? 0
    if (source.ratingNormalized && w > 0) {
      weightedSum += source.ratingNormalized * w
      totalWeight += w
    }
  }

  // 리뷰 볼륨 점수 (100개 이상 = 5점)
  const totalReviews = sources.reduce((sum, s) => sum + (s.reviewCount ?? 0), 0)
  const volumeScore = Math.min(5, (totalReviews / 100) * 5)
  weightedSum += volumeScore * WEIGHTS.reviewVolume
  totalWeight += WEIGHTS.reviewVolume

  const score = totalWeight > 0
    ? Math.round((weightedSum / totalWeight) * 10) / 10
    : 0

  return { score: Math.max(1.0, Math.min(5.0, score)), breakdown: { ... } }
}
```

## 4. 어드민 UI 설계

### 4.1 크롤 모니터 페이지 (`/admin/crawl-monitor`) — 현재 구현됨 + 개선 필요

**현재 구현된 화면:**
- KPI 카드 4개 (총 수집 숙소, 최근 배치, 진행중 작업, 마지막 크롤링)
- 지역별 수집 현황 (바 차트)
- 최근 작업 이력 테이블
- 진행 중 작업 표시

**추가 필요 (Gap):**
```
[수집 시작] 버튼 영역
┌─────────────────────────────────────────────────┐
│  소스 선택: [네이버▼]  지역: [강원도▼]           │
│  [▶ 수집 시작]  [⟳ 전체 재수집]                 │
└─────────────────────────────────────────────────┘
```

### 4.2 인텔리전스 DB 페이지 (`/admin/intelligence`) — 현재 구현됨 + 개선 필요

**현재 구현된 화면:**
- 검색 + 지역 필터 + 영입 여부 필터
- 숙소 카드 목록 (VINTEE Score, 리뷰수, 자동태그)
- 페이지네이션

**추가 필요 (Gap):**
- VINTEE Score 기준 정렬
- 개별 Score 재계산 버튼
- 영입 처리 버튼 및 노트 입력

### 4.3 신규 페이지: 수집 상세 설정 (`/admin/data-collection`)

```
┌──────────────────────────────────────────────────────┐
│  데이터 수집 설정                                      │
├──────────────────────────────────────────────────────┤
│  [크롤 모니터 →]  [인텔리전스 DB →]                   │
├──────────────────────────────────────────────────────┤
│  ▣ VINTEE Score 가중치 설정                           │
│  네이버: 30%  구글: 30%  리뷰볼륨: 20%               │
│  야놀자: 10%  여기어때: 10%                           │
│                          [저장]                       │
├──────────────────────────────────────────────────────┤
│  ▣ API 키 상태                                        │
│  Google Places API: ✅ 설정됨                         │
│  Kakao Local API:   ✅ 설정됨                         │
│  Naver Search API:  ⚠️ 미설정                        │
└──────────────────────────────────────────────────────┘
```
